最大爬坡度

代码：

n=600:4000
i0=5.83,nT=0.85,r=0.367,CDA=2.77,f=0.013,W=38024,nT=0.85
ig=[6.09 3.09 1.71 1.00]
%由于一档时爬坡率最大，令ig取一档值
i=1
Ttq=-19.313+(295.7*n/1000)-(165.44*n.^2/(1000.^2))+(40.874*n.^3/(1000.^3))-(3.8445*n.^4/(1000.^4));
ua=0.377.*r.*n/ig(1,i)/i0;
Ft=(Ttq*ig(1,i)*i0*nT)/r;
Fw=CDA/21.15*ua.^2;
Ff=W*f;
F=Fw+Ff;
%绘制爬坡率与速度关系图，求得最大爬坡率 !此处因坡度较大不可约等
a=asin((Ft-Ff-Fw)/W);
I=tan(a);
Imax=max(tan(a));
plot(ua,I);
xlabel ('ua/(km/h)');
ylabel('i/%');

结果

Imax =

  0.394619999376042

汽车行驶加速度倒数曲线图

代码：

clear all;
%输入原始参数
n=600:4000
i0=5.83,nT=0.85,r=0.367,CDA=2.77,f=0.013,m=3880,W=38024,nT=0.85,a=1.947,hg=0.9,L=3.2,Iw1=1.798,Iw2=3.598,If=0.218;
ig=[6.09 3.09 1.71 1.00]
%for循环绘制驱动力与行驶阻力平衡图
for i = 1:4
Ttq=-19.313+(295.7*n/1000)-(165.44*n.^2/(1000.^2))+(40.874*n.^3/(1000.^3))-(3.8445*n.^4/(1000.^4));
ua=0.377.*r.*n/ig(1,i)/i0;
Ft=(Ttq*ig(1,i)*i0*nT)/r;
Fw=CDA/21.15*ua.^2;
Ff=W*f;
delta=1+(Iw1+Iw2)/(m*r^2)+If*ig(1,i).^2*i0^2*nT/(m*r^2);
alpha=(Ft-Ff-Fw)/(delta*m);
alpha1=alpha;
alpha2=alpha~=0;
alpha3=1./alpha(alpha2)
alpha1(alpha2)=alpha3;
plot(ua,alpha1);
hold on;
grid on;
xlabel ('ua/(km/h)');
ylabel('a/(m/s^2)');
end

汽车二档起步加速至70km/h的时间

代码：

clear all;
%输入原始参数
n=600:4000;
i0=5.83,nT=0.85,r=0.367,CDA=2.77,f=0.013,m=3880,W=38024,nT=0.85,a=1.947,hg=0.9,L=3.2,Iw1=1.798,Iw2=3.598,If=0.218;
ig=[6.09 3.09 1.71 1.00];
%计算汽车二档起步加速至70km/h的时间
for i = 1:4
Ttq=-19.313+(295.7*n/1000)-(165.44*n.^2/(1000.^2))+(40.874*n.^3/(1000.^3))-(3.8445*n.^4/(1000.^4));
ua=0.377.*r.*n/ig(1,i)/i0;
Ft=(Ttq*ig(1,i)*i0*nT)/r;
Fw=CDA/21.15*ua.^2;
Ff=W*f;
delta=1+(Iw1+Iw2)/(m*r^2)+If*ig(1,i).^2*i0^2*nT/(m*r^2);
alpha=(Ft-Ff-Fw)/(delta*m);
alpha1=alpha;
alpha2=alpha~=0;
alpha3=1./alpha(alpha2);
alpha1(alpha2)=alpha3;

if i == 2
ua2max = max(ua);
t2 = trapz(ua./3.6,alpha1);
end

if i == 3
ua3max = max(ua);
n3min = round(ua2max/2/pi*60*ig(1,3)*i0/r*1000/3600);
t3=trapz(ua((n3min-600):3400)./3.6,alpha1((n3min-600):3400));
end

if i == 4
ua4max = max(ua);
n4min = round(ua3max/2/pi*60*ig(1,4)*i0/r*1000/3600);
n4max = round(70/2/pi*60*ig(1,4)*i0/r*1000/3600);
t4=trapz(ua((n4min-600):n4max)./3.6,alpha1((n4min-600):n4max));
end

end
%输出总时间
T = t2 + t3 + t4

结果：

T =
 42.1203

第三章

代码：

clear all;
%输入原始参数
n=600:4000;
i0=[5.17 5.43 5.83 6.17 6.33],nT=0.85,r=0.367,CDA=2.77,f=0.013,m=3880,W=38024,nT=0.85,a=1.947,hg=0.9,L=3.2,Iw1=1.798,Iw2=3.598,If=0.218;
ig=[6.09 3.09 1.71 1.00];
%计算各主减速器传动比下汽车二档起步加速至70km/h的时间
for j = 1:5
for i = 1:4
Ttq=-19.313+(295.7*n/1000)-(165.44*n.^2/(1000.^2))+(40.874*n.^3/(1000.^3))-(3.8445*n.^4/(1000.^4));
ua=0.377.*r.*n/ig(1,i)./i0(1,j);
Ft=(Ttq*ig(1,i).*i0(1,j).*nT)/r;
Fw=CDA/21.15*ua.^2;
Ff=W*f;
delta=1+(Iw1+Iw2)/(m*r^2)+If*ig(1,i).^2.*i0(1,j).^2*nT/(m*r^2);
alpha=(Ft-Ff-Fw)/(delta*m);
alpha1=alpha;
alpha2=alpha~=0;
alpha3=1./alpha(alpha2);
alpha1(alpha2)=alpha3;
if i == 2
t0 = ua(1).*alpha1(1);
ua2max = max(ua);
t2 = trapz(ua./3.6,alpha1);
end
if i == 3
ua3max = max(ua);
n3min = round(ua2max/2/pi*60*ig(1,3).*i0(1,j)/r*1000/3600);
t3=trapz(ua((n3min-600):3400)./3.6,alpha1((n3min-600):3400));
end
if i == 4
ua4max = max(ua);
n4min = round(ua3max/2/pi*60*ig(1,4).*i0(1,j)/r*1000/3600);
n4max = round(70/2/pi*60*ig(1,4).*i0(1,j)/r*1000/3600);
t4=trapz(ua((n4min-600):n4max)./3.6,alpha1((n4min-600):n4max));
end
end
%输出总时间（时间越长动力性越差）
T(1,j) = t2 + t3 + t4
end

%绘制燃油经济性-加速时间曲线

i0 = [5.17 543 5.83 6.17 6.33]%主减速器传动比
Qs = [15.77 15.94 16.38 16.89 17.17]%燃油经济性（越大越差）
x=15.77:0.001:17.17
y=polyfit(Qs,T,3)
Y=polyval(y,x)
plot(1./Qs,1./T,'bo')
hold on
plot(1./x,1./Y)%取倒数画图
title('燃油经济性-加速时间曲线')
set(gca,'xtick',x); 
xlabel('燃油经济性') 
set(gca,'ytick',x);
ylabel('动力性')

结论：

由图可以看出，该图曲线为c曲线。曲线表明，i0值较大时加速时间较短但油耗增多；i0值较小时，加速时间变长但燃油经济性改善。若选择i0=5.83则能较好兼顾动力性与燃油经济性。

Ef = ((L-a(1,i))./L)./(beta-varphi.*hg(1,i)./L)
%后轴制动效率
Er = (a(1,i)./L)./((1-beta)+varphi.*hg(1,i)./L)
figure(1)

plot(varphi,Er)
hold on
plot(varphi,Ef)
xlabel('附着系数varphi')

end

4.3.3

代码：

clear all;
%输入原始参数（矩阵中列1为空载，2为满载）
ua0 = 30 , varphi = 0.80 , tau21 = 0.02 , tau22 = 0.02 ,
Er = [0.672,0.8715] , g = 9.8 , m = [4080,9290] , hg = [0.845,1.170] ,
L = 3.950 , a = [2.100,2.950] , beta = 0.38
%制动系前部损坏
% Fxb = G/L*(a-z*hg) = G*z
for i = 1:2
z = a(1,i).*varphi./(L+(varphi.*hg(1,i)))
abmax = z*g
s1(1,i) = 1/3.6.*(tau21+(tau22/2)).*ua0+ua0.^2/(25.92.*abmax)
end

%制动系后部损坏
% Fxb = G/L*(b+z*hg) = G*z
for i = 1:2
z = (L-a(1,i)).*varphi./(L-(varphi.*hg(1,i)))
abmax = z*g
s2(1,i) = 1/3.6.*(tau21+(tau22/2)).*ua0+ua0.^2/(25.92.*abmax)
end

结果：

s1 =

   10.0061    7.5854
   
s2 =

	8.0879   13.5986

即制动系前部损坏时，空载时制动距离为10.01米，满载时为7.59米；

制动系后部损坏时，空载时制动距离为8.09米，满载时为13.60米。

代码：

clear all;
%输入原始参数
m = 1818.2 , Iz = 3885 , L = 3.048 , a = 1.463 , b = 1.585 , k1 = -62618 ,
k2 = -110185 , i = 20 ;
%求稳定性因数K和特征车速uch
K = m/(L^2)*((a/k2)-(b/k1))
uch = (1/K)^0.5
%绘制稳态横摆角速度增益曲线
ua = 1:150 %车速（km/h）
u = ua./3.6 %车速（m/s）
y=(u./L)./(1+(K.*(u.^2)))
figure(1)
plot(ua,y)
title('稳态横摆角速度增益曲线')
xlabel('ua/(km/h)')
ylabel('稳态横摆增益')
%转向灵敏度计算
u1 = 22.35
z=((u1./L)./(1+(K.*(u1.^2))))/i
%计算静态储备系数S.M.;计算R/R0
SM = (k2/(k1+k2))-(a/L)
ay = 0.4*9.8 , R0 = 15
delta = L/R0
R = L/(delta-(K*ay*L))
X = R/R0
%计算固有频率w0；阻尼比xi；反应时间tau；峰值反应时间varepsilon
u2 = 30.56
w0 = L/u2*(k1*k2*(1+K*(u2^2))/m/Iz)^0.5
xi = (-m*((a^2)*k1+(b^2)*k2)-Iz*(k1+k2))/(2*L*(m*Iz*k1*k2*(1+(K*(u2^2))))^0.5)
tau = atan(((1-(xi)^2)^0.5)/((-w0*m*u2*a/(L*k2))-xi))/(w0*(1-(xi^2))^0.5)
varepsilon = atan(((1-(xi)^2)^0.5)/xi)/(w0*(1-(xi^2))^0.5)+tau

结果：

（1）

  • 稳定性因数

    $$K=0.0024s^{2}/m^{2}$$

  • 特征车速
    $$u_{ch}=20.6053km/h$$

（2）

  • 转向灵敏度为
    $$0.1685$$

（3）

  • 静态储备系数
    $$S.M.=0.1576$$

    $$R/R_0=1.1608$$

（4）

  • 固有频率

    $$\omega _0=5.5758$$

  • 阻尼比

    $$\xi =0.5892$$

  • 反应时间

    $$\tau=0.1811s$$
